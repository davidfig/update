    function testRandom()

    function testPI()
    {
        var test = Math.pow(Math.PI, 100);
        return test;
    }

    function testDelay()
    {
        var test = 0;
        for (var i = 0; i < 100000; i++)
        {
            test += i * 3;
        }
        return test;
    }

    Debug.init();
    var random = Debug.add('random', {text: '0'});
    Update.init();
    Update.add(testRandom, 100, {percent: 'Random Numbers'});
    Update.add(testPI, null, {percent: 'PI Calculation'});
    Update.add(testDelay);
    Update.update();
